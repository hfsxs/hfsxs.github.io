Linux系统离线下载工具Aria2的安装与配置

1.安装Aria2

sudo apt install -y aria2
sudo yum install -y aria2

2.创建配置文件

sudo mkdir -p /etc/aria2
sudo vi /etc/aria2/.conf


# 设置下载文件存储路径, 默认为Download
dir=/home/backups/Downloads
# 设置日志级别
log-level=warn
# 设置日志存储路径
log=/home/backups/Downloads/.logs/aria2.log
# 启用磁盘缓存, 0为禁用, 默认为16M
# disk-cache=32M
# 设置文件预分配方式以降低磁盘碎片, 默认为prealloc
# NTFS建议为falloc, EXT3/4建议为trunc
# file-allocation=none
# 开启断点续传功能
continue=true
# 设置下载任务数, 运行时可修改, 默认为5
max-concurrent-downloads=10
# 设置服务器连接数, 默认为1
max-connection-per-server=5
# 设置文件分片大小, 范围1M-1024M, 默认为20M
min-split-size=50M
# 设置单个任务最大线程数, 默认为5
split=20
# 设置服务器下载速度限制, 运行时可修改, 默认为0
# max-overall-download-limit=0
# 设置单个任务下载速度限制, 默认为0
# max-download-limit=0
# 设置服务器上传速度限制, 运行时可修改, 默认为0
max-overall-upload-limit=1M
# 设置单个任务上传速度限制, 默认为0
# max-upload-limit=1000
# 设置是否禁用IPv6, 默认为false
# disable-ipv6=false
# 设置会话文件路径，下载时先从中读取下载任务
input-file=/home/backups/Downloads/.logs/aria2.session
# 设置会话文件路径，用于服务退出时保存错误或未完成的下载任务
save-session=/home/backups/Downloads/.logs/aria2.session
# 设置回话文件保存时间间隔，默认为0，表示退出再保存 
save-session-interval=60
# 设置是否启用RPC服务, 默认为false
enable-rpc=true
# 设置允许所有RPC来源, 默认为false
rpc-allow-origin-all=true
# 设置允许非外部访问, 默认为false
rpc-listen-all=true
# 设置事件轮询方式, 默认值跟随操作系统
# event-poll=select
# 设置RPC端口, 默认为6800
rpc-listen-port=6800
# 设置RPC授权令牌
rpc-secret=Aria2_2020
# 设置是否启用RPC服务SSL/TLS加密，需要SSL证书
# rpc-secure=true
# 设置RPC服务SSL证书路径
# rpc-certificate=/etc/nginx/ssl/nginx.crt
# 设置RPC服务SSL证书私钥文件路径
# rpc-private-key=/etc/nginx/ssl/nginx.crt
# 设置种子文件下载(.torrent)完成自动开始BT任务, 默认为true
follow-torrent=true
# 设置BT端口,  取值范围6881-6999
listen-port=51413
# 设置单个种子最大连接数, 默认为55
# bt-max-peers=55
# 设置是否启用DHT功能, PT需禁用，默认为true
enable-dht=true
# 设置是否开启IPv6 DHT功能, PT需禁用
# enable-dht6=false
# 设置DHT网络端口, 取值范围6881-6999
# dht-listen-port=6881-6999
# 设置是否开启本地节点查找功能，PT需禁用，默认为false
# bt-enable-lpd=true
# 设置开启种子交换功能, PT需禁用, 默认为true
enable-peer-exchange=true
# 设置单个种子限速, 默认为50K，用于少种的PT
# bt-request-peer-speed-limit=50K
# 设置客户端伪装, PT所需
peer-id-prefix=-TR2770-
# 设置伪装的客户端版本
user-agent=Transmission/2.77
# 设置停止做种的分享率，0为一直做种, 默认为1.0
seed-ratio=0.0001
# 设置最小做种时间，单位分钟
seed-time=1
# 设置强制保存会话功能, 即已完成的任务也保留回话文件, 默认为false
force-save=true
# 设置是否开启BT校验, 默认为true
# bt-hash-check-seed=true
# 设置BT任务重新开始再次校验, 默认为false
bt-seed-unverified=true
# 设置是否将磁力链接元数据保存为种子文件, 默认为false
# bt-save-metadata=true

3.创建启动脚本

sudo vi /lib/systemd/system/aria2.service


[Unit]
Description=Aria2
After=sshd.service

[Service]
PIDFile=/run/aria2.pid
ExecStart=/usr/bin/aria2c -c --conf-path=/etc/aria2/aria2.conf

[Install]
WantedBy=multi-user.target

4.启动Aria2

sudo systemctl daemon-reload
sudo systemctl start aria2.service
sudo systemctl enable aria2.service
